>> LIKE OPERATOR & WILD CARD
---------------------------------------------------------------------------------------------------
SELECT DATA WITH LIKE OPERATOR
----------------------------------------------
WHERE NAME LIKE "%S"
WHERE NAME LIKE "S%"
---------------------------------------------------------
% MEANS CHARACTER SHOULD END FROM S.

S MEANS CHARACTER SHOULD START FROM S.
-------------------------------------------------------------------------------------
WILD CARD CHARACTERS 
-----------------------------------------------------------
% -> MODULUS [PERCENTAGE] -  REPRESENT ZERO, ONE, OR MULTIPLE CHARACTERSS.

_ -> UNDERSCORE - REPRESENT A SINGLE CHARACTERS
--------------------------------------------------------------------------------------
PATTERN 	| DESCRPTIONS
--------------------------------------------------------------------------------------------
LIKE "%a"	|"End form a"
--------------------------------------
LIKE "a%"	| "starts from a"
------------------------------------------------
LIKE "%AB%"	| "HAVE 'AB' ANY WHERE IN CONTENT OR WORDS "
------------------------------------------------------------------
LIKE "A%B"	| "START WITH 'A' AND END WITH 'B' "
-----------------------------------------------------------------------------------
LIKE "_A%"	| SECOND POSITION OF CHARACTER 'A'
----------------------------------------------------------------------------
LIKE "_ _A%" 	| THIRD POSITION OF CHARACTER 'A'
-------------------------------------------------------------------------
LIKE "_OY"	| 'O' IN THE SECOND POSITION AND 'Y' IN THE THIRD POSITION.
-----------------------------------------------------------------------------------

SYNTAX:
	SELECT COLUMN1,COLUMN2,COLUMN3...........
	FROM TABLE_NAME
	WHERE COLUMN_NAME LIKE PATTERN
--------------------------------------------------------------------------------------------
LIKE WITH NOT OPERATOR
------------------------------------------------------------------------------------------
SELECT COLUMN1, COLUMN2, COLUMN3.......
FROM TABLE_NAME
WHERE COLUMN_NAME NOT LIKE PATTERN;
---------------------------------------------------------------------------------------------

>> MYSQL REGULAR EXPRESSION
-------------------------------------------------------------------------------------------
SELECT DATA WITH REGULAR EXPRESSION
---------------------------------------------------------------------------
>> WE FOUND MULTIPLE RESULT USING REGULAR EXPRESSION [RegExp]
----------------------------------------------------------------------------------
EX:-
	WHERE NAME REGEXP "KHAN$ | POOR"
--------------------------------------------------------------------------------------------
$ -> DOLLAR MEANS - IT DENOTED LAST CHARACTER.
| -> NEXT CHARACTER
------------------------------------------------------------------------------------------
SIGN	| PATTERN	| DESCRIPTION
-----------------------------------------------------------------------------------------
^[caret]| '^ra'		| Beginning of string.
------------------------------------------------------------
$[dollar]| "an$"	| End of string value.
-----------------------------------------------------------------------------------
[....]|	"[rms]"		| Any character listed between the square bracket.
----------------------------------------------------------------------------------
^[...]|	"^[rms]"	| Begins with any character listed between the square bracket.
------------------------------------------------------------------------------------
[a-z]|	"[a-z]e"	| starts from a and e is the second position char i.e- ae,be,ce,de,ee,fe.....etc.
------------------------------------------------------------------------------------
p1|p2|p3|"tom|harry|jerry"| match any of the patters like p1,p2 or p3
----------------------------------------------------------------------------------------
syntax:
-------------------------------
select column1, column2, column3
from table_name
where column_name REGEXP PATTERN;

SELECT COLUMN_NAME FROM TABLE1
UNION/ UNION ALL
SELECT * FROM TABLE2;
--------------------------------------------------------------------------------------------------------------------


RULE:
-----
>> EACH SELECT STATMENT WITHIN UNION MUST HAVE THE SAME NUMBER OF COLUMNS.
>> THE COLUMNS MUST ALSO HAVE SIMILAR DATA TYPE.
>> THE COLUMN IN EACH SELECT STATMENT MUST ALSO BE IN THE SAME ORDER.
--------------------------------------------------------------------------------------------------
